            moneyTotal = moneyTotal + rublesAMonth; // –î–æ–±–∞–≤–∏–ª–∏ —ç—Ç–∏ –¥–µ–Ω—å–≥–∏ –∫ –æ—Ç–¥–∞–Ω–Ω—ã–º
            monthCount = monthCount + 1; // –ó–∞—Å—á–∏—Ç–∞–ª–∏ –º–µ—Å—è—Ü

    }
